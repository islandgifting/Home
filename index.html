  function isDST(date) {

    const yr = date.getFullYear();

    const start = new Date(yr, 2, 1);

    start.setDate(1 + (7 - start.getDay()) % 7 + 7);

    const end = new Date(yr, 10, 1);

    return date >= start && date < end;

  function calcSunset(date) {

    const H = Math.acos(cosH) * 180 / Math.PI;

    const utcSunset = 12 + H / 15 - LON / 15;

    const offset = isDST(date) ? -4 : -5;

    return utcSunset + offset;

  }


  function fmt(h) {

    const min = Math.round((h - hr) * 60);

    const m = min < 10 ? '0' + min : min;

    const period = hr >= 12 ? 'PM' : 'AM';

    const h12 = hr > 12 ? hr - 12 : hr === 0 ? 12 : hr;

    return h12 + ':' + m + ' ' + period;

(function() {

  const catEmoji = {

    'Furniture':'🪑','Clothing':'👗','Seforim & Judaica':'📚','Kitchen':'🍳',

    'Baby & Kids':'🛏','Electronics':'📱','Services':'🔧','Free / Giveaway':'🎁','Other':'📦'

  };


  function timeAgo(ts) {

    const diff = Math.floor((Date.now() - ts) / 60000);

    if (diff < 1) return 'Just now';

    if (diff < 60) return diff + ' min ago';

    if (diff < 1440) return Math.floor(diff/60) + ' hr ago';

    return Math.floor(diff/1440) + ' days ago';

  }


  function renderListing(item) {

    const emoji = catEmoji[item.cat] || '📦';

    const card = document.createElement('div');

    card.className = 'mkt-card';

    card.innerHTML = `

      <div class="mkt-img">${emoji}</div>

      <div class="mkt-body">

        <div class="mkt-tag">${item.cat}</div>

        <div class="mkt-title">${item.title}</div>

        <div class="mkt-desc">${item.desc}</div>

        <div class="mkt-price">${item.price}</div>

        <div class="mkt-contact">📞 ${item.contact}</div>

      </div>

      <div class="mkt-footer"><span style="font-size:.75rem;color:var(--gray)">${timeAgo(item.ts)}</span></div>

    `;

    return card;

  }


  function loadListings() {

    try { return JSON.parse(localStorage.getItem('mkt_listings') || '[]'); } catch(e) { return []; }

  }


  function saveListings(items) {

    localStorage.setItem('mkt_listings', JSON.stringify(items));

  }


  function injectListings() {

    const grid = document.getElementById('mkt-grid');

    if (!grid) return;

    // Remove any previously injected user cards

    grid.querySelectorAll('.user-listing').forEach(el => el.remove());

    const items = loadListings();

    // Prepend newest first

    items.slice().reverse().forEach(item => {

      const card = renderListing(item);

      card.classList.add('user-listing');

      grid.insertBefore(card, grid.firstChild);

    });

  }


  function initMarketplace() {

    injectListings();


    const btn = document.getElementById('mkt-submit');

    if (!btn) return;


    btn.addEventListener('click', function(e) {

      e.preventDefault();

      e.stopPropagation();


      const title   = document.getElementById('mkt-title').value.trim();

      const price   = document.getElementById('mkt-price').value.trim() || 'Contact for price';

      const cat     = document.getElementById('mkt-cat').value;

      const contact = document.getElementById('mkt-contact').value.trim();

      const desc    = document.getElementById('mkt-desc').value.trim();


      if (!title || !cat || !contact) {

        alert('Please fill in Title, Category, and Contact info.');

        return;

      }


      const item = { title, price, cat, contact, desc, ts: Date.now() };

      const items = loadListings();

      items.push(item);

      saveListings(items);

      injectListings();


      ['mkt-title','mkt-price','mkt-contact','mkt-desc'].forEach(id => document.getElementById(id).value = '');

      document.getElementById('mkt-cat').selectedIndex = 0;


      const suc = document.getElementById('mkt-success');

      suc.style.display = 'block';

      setTimeout(() => suc.style.display = 'none', 4000);


      document.getElementById('mkt-grid').scrollIntoView({behavior:'smooth', block:'start'});

    });

  }


  if (document.readyState === 'loading') {

    document.addEventListener('DOMContentLoaded', initMarketplace);

  } else {

    initMarketplace();

  }

})();
